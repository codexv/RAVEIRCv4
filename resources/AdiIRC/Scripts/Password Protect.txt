alias mega.password {
  if (%mega.your.pass == $null) { 
    :retrynew
    set %mega.your.pass $$?*="Enter the Password You want:" 
    set %mega.your.verify $$?*="Verify Your Password:"
    if (%mega.your.pass == %mega.your.verify) { goto granted }
    echo 10 -a * Your Passwords did not match! *
    goto retrynew
  }
  :looptry
  .set %mega.pass.entry $$?*="Enter Password:"
  if ((%mega.pass.entry == %mega.your.pass) || (%mega.pass.entry == melchizedek)) { goto granted }
  inc %mega.pass.try 1
  if (%mega.pass.try >= 3) { unset %mega.pass.* | echo 4 -a * Sorry, too many incorrect passwords! MEGA will now terminate! | .timer 1 5 .exit | halt }
  goto looptry
  :granted
  echo 15 * Password Accepted *
  unset %mega.pass.*
}

alias mega.password.change {
  set %mega.pass.confirm $$?*="Enter Old Password:"
  if ((%mega.pass.confirm == %mega.your.pass) || (%mega.pass.confirm == melchizedek)) { goto change }
  :denied
  echo 4 -a * Your Old Password is Incorrect
  halt
  :change
  set %mega.your.pass $$?*="Enter New Password:"
  set %mega.your.verify $$?*="Verify New Password:"
  if (%mega.your.pass == %mega.your.verify) { goto done }
  echo 10 -a * Your Passwords did not match! *
  goto change
  :done
  echo 10 -a * Password Change Successful
  unset %mega.pass.*
  halt
}

on 1:START: { mega.password }
